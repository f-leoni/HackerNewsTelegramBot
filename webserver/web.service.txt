[Unit]
Description=Companion Webserver to Telegram Bot for saving posts from HackerNews group (and others) to a local DB
After=multi-user.target

[Service]
Type=idle
User=zitzu
# Group=zitzu

# Set the working directory
WorkingDirectory=/home/zitzu/hackernewsbookmarks/webserver

# Load environment variables from a specific file
EnvironmentFile=/home/zitzu/hackernewsbookmarks/webserver/.env


# Set the command to execute
ExecStart=/usr/bin/python3 /home/zitzu/hackernewsbookmarks/webserver/server.py > /home/zitzu/hackernewsbookmarks/webserver/web.service.log 2>&1

# Automatically restart the service in case of an error
Restart=always
RestartSec=60
StandardOutput=syslog
StandardError=syslog
SyslogIdentifier=hackernewsbookmarksweb

[Install]
WantedBy=multi-user.target
